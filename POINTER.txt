POINTER

#include <iostream>
using namespace std;
int main() {
    
    int a = 10;
    //Declaration
    int *ptr ;
    ptr = &a;
    
    cout << "a = " << a << endl;
    cout << "Address of a = " << &a << endl;
    cout << "Address of a through pointer = " << ptr << endl;
    cout << "*ptr = " << *ptr << endl;
    
    return 0;
}
