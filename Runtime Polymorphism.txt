#include <iostream>
using namespace std;

class Animal {
public:
    virtual void sound() {
        cout << "Animal makes sound" << endl; 
    }
};
class Dog : public Animal 
    {
public:
    void sound() {  //overide base class function
         cout << "Dog barks" << endl;
        }
};

int main() {
    
    Animal* a;   //base class pointer
    Dog d;
    a = &d;
    a->sound();   //Output: Animal makes sound

    return 0;
}
