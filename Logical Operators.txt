#include <iostream>
using namespace std;
int main() {
    
    int a=10,b=20;
  
    cout << "logical AND: " << (a<0 && b>0) << endl;
    cout << "logical OR : " << (a>0 || b<0)<< endl;
    cout << "logical NOT : " << (a>0)<< endl;
    
    return 0;
}
