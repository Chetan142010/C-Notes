#include <iostream>
 using namespace std;
 
 class car{
     public:
     string make;
     string model;
     int year;
     
     //default constructions ( no parameters)
     
     car(){
         make = "BENZ";
         model = "Xo8";
         year = 2019;
         
     }
     void displayinfo(){
         cout << year <<" "<< make <<" "<<model <<endl;
     }
 };
  int main(){
      car myCar;
      
      myCar.displayinfo();
      return 0;
  }
